    15	    """Validates and sanitizes file paths to prevent path traversal attacks."""
    16	    
    17	    def __init__(self, allowed_base_paths: list[str]):
    18	        """
    19	        Initialize path validator with allowed base paths.
    20	        
    21	        Args:
    22	            allowed_base_paths: List of absolute paths that are allowed as base directories
    23	        """
    24	        self.allowed_base_paths = [os.path.abspath(p) for p in allowed_base_paths]
    25	    
    26	    def validate_path(self, path: str) -> str:
    27	        """
    28	        Validate that a path is safe and within allowed directories.
    29	        
    30	        Args:
    31	            path: Path to validate
    32	        
    33	        Returns:
    34	            Absolute, normalized path if valid
    35	        
    36	        Raises:
    37	            PathValidationError: If path is invalid or outside allowed directories
    38	        """
    39	        if not path:
    40	            raise PathValidationError("Path cannot be empty")
    41	        
    42	        # Resolve to absolute path and normalize
    43	        try:
    44	            abs_path = os.path.abspath(path)
    45	            real_path = os.path.realpath(abs_path)
    46	        except (OSError, ValueError) as e:
    47	            raise PathValidationError(f"Invalid path: {e}")
    48	        
    49	        
    50	        # Verify path is within allowed base paths
    51	        is_allowed = False
    52	        for base_path in self.allowed_base_paths:
    53	            try:
    54	                # Check if real_path is within base_path
    55	                Path(real_path).relative_to(base_path)
    56	                is_allowed = True
    57	                break
    58	            except ValueError:
    59	                continue
    60	        
    61	        if not is_allowed:
    62	            raise PathValidationError(
    63	                f"Path {path} is outside allowed directories: {self.allowed_base_paths}"
    64	            )
    65
