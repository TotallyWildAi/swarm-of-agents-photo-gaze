+"""Efficient thumbnail generation with LRU caching and file-hash-based invalidation."""
+import io
+import time
+from functools import lru_cache
+from typing import Optional
+from PIL import Image
+from sqlalchemy.orm import Session
+from app.models import Photo
+
+
+class ThumbnailGenerator:
+    """Generate and cache thumbnails with automatic invalidation on file changes."""
+    
+    def __init__(self, max_cache_size: int = 1000):
+        """Initialize thumbnail generator with LRU cache.
+        
+        Args:
+            max_cache_size: Maximum number of thumbnails to cache in memory
+        """
+        self.max_cache_size = max_cache_size
+        # Cache structure: {(photo_id, size, file_hash): jpeg_bytes}
+        # Using file_hash in key ensures automatic invalidation on file changes
+        self._cache = {}
+        self._cache_order = []  # Track insertion order for LRU eviction
+    
+    async def get_thumbnail(self, photo_id: int, size: int, session: Session) -> Optional[bytes]:
+        """Get or generate thumbnail for a photo.
+        
+        Cache key includes file_hash to detect file modifications.
+        When file_hash changes (detected by folder_scanner), old cache entries
+        are automatically invalidated because the key changes.
+        
+        Args:
+            photo_id: ID of the photo
+            size: Thumbnail size in pixels
+            session: Database session
+        
+        Returns:
+            JPEG thumbnail bytes or None if photo not found
+        """
+        # Fetch photo metadata from database
+        photo = session.query(Photo).filter(Photo.id == photo_id).first()
+        if not photo:
+            return None
+        
+        # Use file_hash in cache key for automatic invalidation on file changes
+        cache_key = (photo_id, size, photo.file_hash)
+        
+        # Check cache first (sub-1ms lookup)
+        if cache_key in self._cache:
+            return self._cache[cache_key]
+        
+        # Generate thumbnail (target: 30-40ms for typical images)
+        start_time = time.time()
+        thumbnail_bytes = await self._generate_thumbnail(photo.file_path, size)
+        elapsed = (time.time() - start_time) * 1000
+        
+        if thumbnail_bytes is None:
+            return None
+        
+        # Store in cache with LRU eviction
+        self._add_to_cache(cache_key, thumbnail_bytes)
+        
+        # Log performance metrics
+        print(f"Generated thumbnail for photo {photo_id} ({size}px) in {elapsed:.1f}ms")
+        
+        return thumbnail_bytes
+    
+    async def _generate_thumbnail(self, file_path: str, size: int) -> Optional[bytes]:
+        """Generate JPEG thumbnail from image file.
+        
+        Uses PIL's efficient thumbnail method which maintains aspect ratio
+        and performs in-place resizing for performance.
+        
+        Args:
+            file_path: Path to the image file
+            size: Thumbnail size in pixels
+        
+        Returns:
+            JPEG thumbnail bytes or None on error
+        """
+        try:
+            with Image.open(file_path) as img:
+                # Convert to RGB if necessary (handles RGBA, grayscale, etc.)
+                if img.mode != 'RGB':
+                    img = img.convert('RGB')
+                
+                # Create thumbnail maintaining aspect ratio
+                # thumbnail() modifies in-place and is faster than resize()
+                img.thumbnail((size, size), Image.Resampling.LANCZOS)
+                
+                # Encode to JPEG bytes with quality optimization
+                output = io.BytesIO()
+                img.save(output, format='JPEG', quality=85, optimize=True)
+                return output.getvalue()
+        except Exception as e:
+            print(f"Error generating thumbnail for {file_path}: {e}")
+            return None
+    
+    def _add_to_cache(self, cache_key: tuple, thumbnail_bytes: bytes) -> None:
+        """Add thumbnail to cache with LRU eviction.
+        
+        When cache exceeds max_cache_size, removes least recently used entry.
+        
+        Args:
+            cache_key: (photo_id, size, file_hash) tuple
+            thumbnail_bytes: JPEG thumbnail data
+        """
+        # If key already exists, remove it to update order
+        if cache_key in self._cache:
+            self._cache_order.remove(cache_key)
+        
+        # Add to cache and track order
+        self._cache[cache_key] = thumbnail_bytes
+        self._cache_order.append(cache_key)
+        
+        # Evict least recently used if cache is full
+        if len(self._cache) > self.max_cache_size:
+            lru_key = self._cache_order.pop(0)
+            del self._cache[lru_key]
+    
+    def clear_cache(self) -> None:
+        """Clear all cached thumbnails (for testing or manual invalidation)."""
+        self._cache.clear()
+        self._cache_order.clear()
+    
+    def get_cache_stats(self) -> dict:
+        """Get cache statistics for monitoring.
+        
+        Returns:
+            Dictionary with cache size and hit/miss info
+        """
+        return {
+            "cached_thumbnails": len(self._cache),
+            "max_cache_size": self.max_cache_size,
+            "cache_utilization_percent": (len(self._cache) / self.max_cache_size) * 100
+        }
