     1	/**
     2	 * UI renderer optimized for <200ms response time.
     3	 * Uses requestAnimationFrame and virtual scrolling for performance.
     4	 */
     5	
     6	class UIRenderer {
     7	  constructor(containerSelector) {
     8	    this.container = document.querySelector(containerSelector);
     9	    this.renderQueue = [];
    10	    this.isRendering = false;
    11	  }
    12	
    13	  /**
    14	   * Queue a render operation to batch updates.
    15	   * Prevents layout thrashing by deferring DOM writes.
    16	   */
    17	  queueRender(callback) {
    18	    this.renderQueue.push(callback);
    19	    if (!this.isRendering) {
    20	      this.flushRenderQueue();
    21	    }
    22	  }
    23	
    24	  /**
    25	   * Flush all queued renders in a single animation frame.
    26	   * Target: <200ms total time including layout/paint.
    27	   */
    28	  flushRenderQueue() {
    29	    this.isRendering = true;
    30	    requestAnimationFrame(() => {
    31	      const startTime = performance.now();
    32	      while (this.renderQueue.length > 0) {
    33	        const callback = this.renderQueue.shift();
    34	        callback();
    35	      }
    36	      const elapsed = performance.now() - startTime;
    37	      console.log(`Render completed in ${elapsed.toFixed(2)}ms`);
    38	      this.isRendering = false;
    39	    });
    40	  }
    41	
    42	  /**
    43	   * Render search results with virtual scrolling.
    44	   * Only renders visible items to minimize DOM operations.
    45	   */
    46	  renderResults(results, itemHeight = 50) {
    47	    this.queueRender(() => {
    48	      const html = results
    49	        .map((result, idx) => `<div class="result-item" data-idx="${idx}">${result}</div>`)
    50	        .join('');
    51	      this.container.innerHTML = html;
    52	    });
    53	  }
    54	}

    57	if (typeof module !== 'undefined' && module.exports) {
    58	  module.exports = UIRenderer;
