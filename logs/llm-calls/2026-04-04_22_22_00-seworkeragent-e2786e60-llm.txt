     9	class QdrantClient:
    10	    """Client for Qdrant vector database with batch insert and search capabilities."""
    11	    
    12	    def __init__(self, url: str = None, collection_name: str = "embeddings"):
    13	        """Initialize Qdrant client.
    14	        
    15	        Args:
    16	            url: Qdrant server URL (defaults to QDRANT_URL env var or http://localhost:6333)
    17	            collection_name: Name of the collection to use (default: embeddings)
    18	        """
    19	        self.url = url or os.getenv("QDRANT_URL", "http://localhost:6333")
    20	        self.collection_name = collection_name
    21	        self.client = QdrantBaseClient(url=self.url)
    22	    
    23	    def batch_insert(self, points: List[Tuple[str, List[float], Dict]]) -> bool:
    24	        """Insert multiple vectors into Qdrant collection.
    25	        
    26	        Args:
    27	            points: List of tuples (point_id, vector, payload) where:
    28	                - point_id: unique string identifier
    29	                - vector: list of floats (1024-dim)
    30	                - payload: dict with metadata
    31	        
    32	        Returns:
    33	            True if insertion succeeded, False otherwise
    34	        """
    35	        try:
    36	            qdrant_points = [
    37	                PointStruct(
    38	                    id=point_id,
    39	                    vector=vector,
    40	                    payload=payload
    41	                )
    42	                for point_id, vector, payload in points
    43	            ]
    44	            self.client.upsert(
    45	                collection_name=self.collection_name,
    46	                points=qdrant_points
    47	            )
    48	            return True
    49	        except Exception as e:
    50	            print(f"Error during batch insert: {e}")
    51	            return False
    52	    
    53	    def similarity_search(self, query_vector: List[float], limit: int = 10, threshold: float = 0.0) -> List[Dict]:
    54	        """Search for similar vectors in Qdrant collection with optional threshold filtering.
    55	        
    56	        Args:
    57	            query_vector: Query vector (1024-dim list of floats)
    58	            limit: Maximum number of results to return
    59	            threshold: Minimum similarity score to include in results (default: 0.0, range: 0.0-1.0)
    60	        
    61	        Returns:
    62	            List of dicts with keys: id, score, payload, filtered by threshold and ranked by score descending
    63	        """
    64	        try:
    65	            results = self.client.search(
    66	                collection_name=self.collection_name,
    67	                query_vector=query_vector,
    68	                limit=limit
    69	            )
    70	            # Filter results by threshold and convert to dicts (already ranked by score descending from Qdrant)
    71	            filtered_results = [
    72	                {
    73	                    "id": result.id,
    74	                    "score": result.score,
    75	                    "payload": result.payload
    76	                }
    77	                for result in results
    78	                if result.score >= threshold
    79	            ]
    80	            return filtered_results
    81	        except Exception as e:
    82	            print(f"Error during similarity search: {e}")
    83	            return []
