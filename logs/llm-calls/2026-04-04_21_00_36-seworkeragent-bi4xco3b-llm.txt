@app.websocket("/ws/progress/{job_id}")
async def websocket_progress(websocket: WebSocket, job_id: str):
    """WebSocket endpoint for real-time progress updates during photo processing.
    
    Broadcasts progress updates including percentage completion and estimated time remaining.
    """
    await websocket.accept()
    try:
        if job_queue_manager is None:
            await websocket.send_json({"error": "Job queue not initialized"})
            await websocket.close()
            return
        
        # Send progress updates every 100ms while job is active
        while True:
            if job_id in job_queue_manager.active_jobs:
                progress_data = await job_queue_manager.get_progress(job_id)
                await websocket.send_json(progress_data)
            else:
                # Job not found or completed
                await websocket.send_json({"status": "not_found"})
                break
            
            await asyncio.sleep(0.1)  # Update every 100ms
    except WebSocketDisconnect:
        pass  # Client disconnected
    except Exception as e:
        print(f"WebSocket error for job {job_id}: {e}")
        try:
            await websocket.send_json({"error": str(e)})
        except:
            pass

        """List similarity groups with pagination, filtering, and sorting.

                avg_similarity = sum(m['score'] for m in group_members) / len(group_members)

    limit: int = Query(default=20, ge=1, le=100),
    min_similarity: Optional[float] = Query(default=None),
    min_quality: Optional[float] = Query(default=None),
    sort_by: Optional[str] = Query(default="similarity"),
    sort_order: Optional[str] = Query(default="desc"),
):
    """Query similarity groups with pagination, filtering, and sorting."""
    # Validate sort_order — invalid values return 400
    if sort_order not in ("asc", "desc"):
        return JSONResponse(status_code=400, content={"detail": "sort_order must be 'asc' or 'desc'"})

    # Default invalid sort_by to 'similarity'
    if sort_by not in ("similarity", "quality"):
        sort_by = "similarity"

    db = SessionLocal()
    try:
        query = db.query(SimilarityGroup)

        # Apply filters
        if min_similarity is not None:
            query = query.filter(SimilarityGroup.similarity_score >= min_similarity)
        if min_quality is not None:
            query = query.filter(SimilarityGroup.quality_score >= min_quality)

        # Count total matching records (before pagination)
        total = query.count()

        # Apply sorting
        sort_column = SimilarityGroup.similarity_score if sort_by == "similarity" else SimilarityGroup.quality_score
        order_func = desc if sort_order == "desc" else asc
        query = query.order_by(order_func(sort_column))

        # Apply pagination
        groups = query.offset(skip).limit(limit).all()

        # Build response items with photo data and thumbnails
        items = []
        for group in groups:
            # Query photos associated with this group (by group_id)
            photos_data = []
            if group.group_id:
                photo_records = db.query(Photo).filter(
                    Photo.id == group.group_id
                ).all()
                for p in photo_records:
                    thumb = thumbnail_generator.get_thumbnail(p.file_path) if p.file_path else None
                    photos_data.append({
                        "id": p.id,
                        "filename": p.filename,
                        "path": p.file_path,
                        "score": group.similarity_score,
                        "thumbnail": thumb,
                    })

            items.append({
                "group_id": group.group_id if group.group_id else group.id,
                "similarity_score": group.similarity_score,
                "quality_score": group.quality_score,
                "member_count": group.member_count if group.member_count else 0,
                "photos": photos_data,
            })

        has_more = (skip + limit) < total

        return {
            "items": items,
            "total": total,
            "skip": skip,
            "limit": limit,
            "has_more": has_more,
        }
    finally:
        db.close()
