    12	def generate_threshold_examples(
    13	    db: Session,

    22	        db: Database session

    26	    
    27	    Returns:

    53	                limit=max_samples_per_threshold
    54	            )
    55	            
    56	            # Track total matches and collect samples
    57	            total_matches += len(matches)
    58	            for match in matches[:max_samples_per_threshold]:
    59	                sample_matches.append({
