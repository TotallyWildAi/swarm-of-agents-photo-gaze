):
    """List similarity groups with pagination, filtering, and sorting.
    
    Args:
        skip: Number of groups to skip (pagination offset)
        limit: Maximum number of groups to return (1-100)
        sort_by: Sort order - 'similarity' or 'quality'
        sort_order: Sort direction - 'asc' or 'desc'
        min_similarity: Minimum similarity score threshold (0.0-1.0)
        min_quality: Minimum quality score threshold (0.0-1.0)
    
    Returns:
        JSON with 'items' list, pagination metadata, and has_more flag
    
    Raises:
        400: Invalid sort_by or sort_order parameter
        422: Invalid limit parameter (must be 1-100)
    """
    session = SessionLocal()
    try:
        # Validate limit parameter (must be 1-100)
        if limit < 1 or limit > 100:
            return JSONResponse(status_code=422, content={"error": "limit must be between 1 and 100"})
        
        # Default invalid sort_by to 'similarity' instead of rejecting with 400
        if sort_by not in ["similarity", "quality"]:
            sort_by = "similarity"
        
        # Validate sort_order parameter
        if sort_order not in ["asc", "desc"]:
            return JSONResponse(status_code=400, content={"error": "sort_order must be 'asc' or 'desc'"})
        
        # Use SimilarityGroupService to fetch groups with filtering
        from app.similarity_search import SimilarityGroupService
        service = SimilarityGroupService(session)
        groups = service.list_groups(
            skip=0,  # Fetch all for filtering, then paginate
            limit=10000,  # Large limit to get all groups for filtering
            sort_by=sort_by,
            sort_order=sort_order,
            min_similarity=min_similarity,
            min_quality=min_quality
        )
        
        # Apply pagination
        total = len(groups)
        has_more = (skip + limit) < total
        paginated_groups = groups[skip:skip + limit]
        
        # Transform groups to response schema with photos/thumbnails
        items = []
        for group in paginated_groups:
            # Build photos list with thumbnails
            photos = []
            for m in group.get('members', []):
                photo_id = m.get('photo_id')
                # Get thumbnail for this photo (async call)
                thumbnail_bytes = await thumbnail_generator.get_thumbnail(photo_id, 200, session)
                # Convert bytes to base64 string for JSON serialization
                thumbnail_b64 = None
                if thumbnail_bytes:
                    import base64
                    thumbnail_b64 = base64.b64encode(thumbnail_bytes).decode('utf-8')
                photos.append({
                    'id': photo_id,
                    'filename': m.get('filename', ''),
                    'path': m.get('file_path', ''),
                    'score': m.get('score', 0),
                    'thumbnail': thumbnail_b64
                })
            item = {
                'group_id': group.get('group_id'),
                'similarity_score': group.get('avg_similarity', 0),
                'quality_score': group.get('quality_score', 0),
                'member_count': group.get('member_count', 0),
                'photos': photos
            }
            items.append(item)
        
        return JSONResponse(status_code=200, content={
            "items": items,
            "total": total,
            "skip": skip,
            "limit": limit,
            "has_more": has_more
        })

        print(f"Error listing similarity groups: {e}")

    finally:
        session.close()

    Returns:
        JSON with group details including all members with thumbnails
    """
    session = SessionLocal()
    try:
        from app.models import Photo
        
        # Verify group representative exists
        photo = session.query(Photo).filter(Photo.id == group_id).first()
        if not photo:
            return JSONResponse(status_code=404, content={"error": "Group not found"})
        
        similarity_search = SimilaritySearch()
        groups = similarity_search.find_similarity_groups(session)
        
        # Find the group containing this photo
        target_group = None
        for group in groups:
            if any(m['photo_id'] == group_id for m in group['members']):
                target_group = group
                break
        
        if not target_group:
            return JSONResponse(status_code=404, content={"error": "Group not found"})
        
        return JSONResponse(status_code=200, content={
            "group_id": group_id,
            "members": target_group['members'],
            "member_count": len(target_group['members']),
            "average_score": sum(m['score'] for m in target_group['members']) / len(target_group['members'])
        })
    except Exception as e:
        print(f"Error getting group detail: {e}")

    finally:
        session.close()
