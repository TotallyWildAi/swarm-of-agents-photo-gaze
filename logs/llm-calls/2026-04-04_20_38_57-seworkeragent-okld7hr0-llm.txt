     2	 * UI renderer optimized for <200ms response time.
     3	 * Uses requestAnimationFrame and virtual scrolling for performance.
     4	 */
     5	
     6	class UIRenderer {
     7	  constructor(containerSelector) {
     8	    this.container = document.querySelector(containerSelector);
     9	    this.renderQueue = [];
    10	    this.isRendering = false;
    11	  }
    12	
    13	  /**
    14	   * Queue a render operation to batch updates.
    15	   * Prevents layout thrashing by deferring DOM writes.
    16	   */
    17	  queueRender(callback) {
    18	    this.renderQueue.push(callback);
    19	    if (!this.isRendering) {
    20	      this.flushRenderQueue();
    21	    }
    22	  }
    23	
    24	  /**
    25	   * Flush all queued renders in a single animation frame.
    26	   * Target: <200ms total time including layout/paint.
    27	   */
    28	  flushRenderQueue() {
    29	    this.isRendering = true;
    30	    requestAnimationFrame(() => {
    31	      const startTime = performance.now();
    32	      while (this.renderQueue.length > 0) {
    33	        const callback = this.renderQueue.shift();
    34	        callback();
    35	      }
    36	      const elapsed = performance.now() - startTime;
    37	      console.log(`Render completed in ${elapsed.toFixed(2)}ms`);
    38	      this.isRendering = false;
    39	    });
    40	  }
    41	
    42	  /**
    43	   * Render search results with virtual scrolling.
    44	   * Only renders visible items to minimize DOM operations.
    45	   */
    46	  renderResults(results, itemHeight = 50) {
    47	    this.queueRender(() => {
    48	      const html = results
    49	        .map((result, idx) => `<div class="result-item" data-idx="${idx}">${result}</div>`)
    50	        .join('');
    51	      this.container.innerHTML = html;
    52	    });
    53	  }
    54	}

    14	        self._cache_hits = 0

    16	
    17	    def encode(self, texts, batch_size=32):
    18	        """Encode texts to embeddings with caching and batching.
    19	        
    20	        Args:
    21	            texts: List of strings or single string
    22	            batch_size: Batch size for encoding (M1 optimized default)
    23	        
    24	        Returns:
    25	            numpy array of embeddings
    26	        """
    27	        if isinstance(texts, str):
    28	            texts = [texts]
    29	        
    30	        # Check cache for exact matches
    31	        uncached = []
    32	        uncached_indices = []
    33	        embeddings = [None] * len(texts)
    34	        
    35	        for i, text in enumerate(texts):
    36	            if text in self._embedding_cache:
    37	                embeddings[i] = self._embedding_cache[text]
    38	                self._cache_hits += 1
    39	            else:
    40	                uncached.append(text)
    41	                uncached_indices.append(i)
    42	                self._cache_misses += 1
    43	        
    44	        # Encode uncached texts in batches
    45	        if uncached:
    46	            new_embeddings = self.model.encode(

    48	                batch_size=batch_size,
    49	                convert_to_numpy=True,
    50	                show_progress_bar=False
    51	            )
    52	            
    53	            for i, idx in enumerate(uncached_indices):
    54	                embeddings[idx] = new_embeddings[i]
    55	                self._embedding_cache[uncached[i]] = new_embeddings[i]
    56	        
    57	        return np.array(embeddings)
    58

    61	        total = self._cache_hits + self._cache_misses
