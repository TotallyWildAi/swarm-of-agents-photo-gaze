async def list_similarity_groups(
    skip: int = 0,

    sort_by: str = "similarity",

    min_similarity: float = 0.0,
    min_quality: float = 0.0
):
    """List similarity groups with pagination, filtering, and sorting.
    
    Args:
        skip: Number of groups to skip (pagination offset)
        limit: Maximum number of groups to return (1-100)
        sort_by: Sort order - 'similarity' or 'quality'
        sort_order: Sort direction - 'asc' or 'desc'
        min_similarity: Minimum similarity score threshold (0.0-1.0)
        min_quality: Minimum quality score threshold (0.0-1.0)
    
    Returns:
        JSON with 'items' list, pagination metadata, and has_more flag
    
    Raises:
        400: Invalid sort_by or sort_order parameter
        422: Invalid limit parameter (must be 1-100)
    """
    session = SessionLocal()
    try:
        # Validate limit parameter (must be 1-100)
        if limit < 1 or limit > 100:
            return JSONResponse(status_code=422, content={"error": "limit must be between 1 and 100"})
        
        # Default invalid sort_by to 'similarity' instead of rejecting with 400
        if sort_by not in ["similarity", "quality"]:
            sort_by = "similarity"
        
        # Validate sort_order parameter
        if sort_order not in ["asc", "desc"]:
            return JSONResponse(status_code=400, content={"error": "sort_order must be 'asc' or 'desc'"})
        
        # Use SimilarityGroupService to fetch groups with filtering
        from app.similarity_search import SimilarityGroupService
        service = SimilarityGroupService(session)
        groups = service.list_groups(
            skip=0,  # Fetch all for filtering, then paginate
            limit=10000,  # Large limit to get all groups for filtering
            sort_by=sort_by,
            sort_order=sort_order,
            min_similarity=min_similarity,
            min_quality=min_quality
        )
        
        # Apply pagination
        total = len(groups)
        has_more = (skip + limit) < total
        paginated_groups = groups[skip:skip + limit]
        
        # Transform groups to response schema with photos/thumbnails
        items = []
        for group in paginated_groups:
            # Build photos list with thumbnails
            photos = []
            for m in group.get('members', []):
                photo_id = m.get('photo_id')
                # Get thumbnail for this photo (async call)
                thumbnail_bytes = await thumbnail_generator.get_thumbnail(photo_id, 200, session)
                # Convert bytes to base64 string for JSON serialization
                thumbnail_b64 = None
                if thumbnail_bytes:
                    import base64
                    thumbnail_b64 = base64.b64encode(thumbnail_bytes).decode('utf-8')
                photos.append({
                    'id': photo_id,
                    'filename': m.get('filename', ''),
                    'path': m.get('file_path', ''),
                    'score': m.get('score', 0),
                    'thumbnail': thumbnail_b64
                })
            item = {
                'group_id': group.get('group_id'),
                'similarity_score': group.get('avg_similarity', 0),
                'quality_score': group.get('quality_score', 0),
                'member_count': group.get('member_count', 0),
                'photos': photos
            }
            items.append(item)
        
        return JSONResponse(status_code=200, content={
            "items": items,
            "total": total,
            "skip": skip,
            "limit": limit,
            "has_more": has_more
        })
    except Exception as e:
        print(f"Error listing similarity groups: {e}")
        return JSONResponse(status_code=500, content={"error": str(e)})
    finally:
        session.close()

    """Get detailed information about a specific similarity group.
    
    Args:

    Returns:
        JSON with group details including all members with thumbnails
    """
    session = SessionLocal()
    try:
        from app.models import Photo
        
        # Verify group representative exists
        photo = session.query(Photo).filter(Photo.id == group_id).first()
        if not photo:
            return JSONResponse(status_code=404, content={"error": "Group not found"})
        
        similarity_search = SimilaritySearch()
        groups = similarity_search.find_similarity_groups(session)
        
        # Find the group containing this photo
        target_group = None
        for group in groups:
            if any(m['photo_id'] == group_id for m in group['members']):
                target_group = group
                break
        
        if not target_group:
            return JSONResponse(status_code=404, content={"error": "Group not found"})
        
        return JSONResponse(status_code=200, content={
            "group_id": group_id,
            "members": target_group['members'],
            "member_count": len(target_group['members']),
            "average_score": sum(m['score'] for m in target_group['members']) / len(target_group['members'])
        })
    except Exception as e:
        print(f"Error getting group detail: {e}")
        return JSONResponse(status_code=500, content={"error": str(e)})
    finally:
        session.close()

@app.post("/rescan")
async def rescan_folder(folder_path: str = None):
    """Trigger manual folder re-scan with change detection and incremental processing.
    
    Scans folder for new/modified/deleted photos, queues changes for processing.
    Returns job_id for tracking progress via WebSocket.
    """
    if job_queue_manager is None:
        return JSONResponse(status_code=503, content={"error": "Job queue not initialized"})
    
    if not folder_path:
        folder_path = os.getenv("PHOTOS_FOLDER", "./photos")
    
    if not os.path.isdir(folder_path):
        return JSONResponse(status_code=400, content={"error": f"Folder not found: {folder_path}"})
    
    try:
        # Initialize scanner and database session
        scanner = FolderScanner()
        session = job_queue_manager.SessionLocal()
        
        # Scan folder for changes (new, modified, deleted photos)
        photo_ids, change_count = scanner.scan_folder(folder_path, session)
        session.close()
        
        if change_count == 0:
            return JSONResponse(status_code=200, content={
                "message": "No changes detected",
                "changes_found": 0
            })
        
        # Create job for processing changed photos
        job_id = str(uuid.uuid4())
        job_created = job_queue_manager.create_job(job_id, change_count)
        
        if not job_created:
            return JSONResponse(status_code=500, content={"error": "Failed to create processing job"})
        
        # Queue photos for processing
        for photo_id in photo_ids:
            asyncio.create_task(job_queue_manager.process_photo(job_id, photo_id))
        
        return JSONResponse(status_code=202, content={
            "job_id": job_id,
            "message": "Rescan initiated",
            "changes_found": change_count,
            "photos_queued": len(photo_ids)
        })
    except Exception as e:
        print(f"Error during rescan: {e}")
        return JSONResponse(status_code=500, content={"error": str(e)})


@app.websocket("/ws/progress/{job_id}")
async def websocket_progress(websocket: WebSocket, job_id: str):
    """WebSocket endpoint for real-time progress updates during photo processing.
    
    Broadcasts progress updates including percentage completion and estimated time remaining.
    """
    await websocket.accept()
    try:
        if job_queue_manager is None:
            await websocket.send_json({"error": "Job queue not initialized"})
            await websocket.close()
            return
        
        # Send progress updates every 100ms while job is active
        while True:
            if job_id in job_queue_manager.active_jobs:
                progress_data = await job_queue_manager.get_progress(job_id)
                await websocket.send_json(progress_data)
            else:
                # Job not found or completed
                await websocket.send_json({"status": "not_found"})
                break
            
            await asyncio.sleep(0.1)  # Update every 100ms
    except WebSocketDisconnect:
        pass  # Client disconnected
    except Exception as e:
        print(f"WebSocket error for job {job_id}: {e}")
        try:
            await websocket.send_json({"error": str(e)})
        except:
            pass

from typing import List, Dict, Optional

class SimilarityGroupService:
    """Service for querying similarity groups from Qdrant with pagination and filtering."""
    
    def __init__(self, db: Session):
        """Initialize service with database session and Qdrant client.
        
        Args:
            db: SQLAlchemy database session
        """
        self.db = db
        self.qdrant_client = QdrantClient()
    
    def list_groups(
        self,
        skip: int = 0,
        limit: int = 10,
        sort_by: str = "similarity",
        sort_order: str = "desc",
        min_similarity: float = 0.0,
        min_quality: float = 0.0
    ) -> List[Dict]:
        """List similarity groups with pagination, filtering, and sorting.
        
        Args:
            skip: Number of groups to skip (pagination offset)
            limit: Maximum number of groups to return
            sort_by: 'similarity' (by score) or 'quality' (by resolution/file_size)
            sort_order: 'asc' or 'desc' for sort direction
            min_similarity: Minimum similarity score threshold (0.0-1.0)
            min_quality: Minimum quality score threshold (0.0-1.0)
        
        Returns:
            List of similarity groups with metadata and thumbnails
        """
        # Fetch all photos with embeddings from database
        # Filter by min_similarity at query level for efficiency
        photos = self.db.query(Photo).all()
        if not photos:
            return []
        
        # Build groups by querying Qdrant for each photo's embedding
        groups = []
        processed_ids = set()
        
        for photo in photos:
            if photo.id in processed_ids:
                continue
            
            # Get embedding for this photo
            embedding = self.db.query(Embedding).filter(
                Embedding.photo_id == photo.id
            ).first()
            
            if not embedding or not embedding.qdrant_point_id:
                continue
            
            # Query Qdrant for similar photos (similarity group)
            # Note: In production, would fetch actual vector from Qdrant
            # For now, use a placeholder that would be replaced with real vector
            group_members = self._get_group_members(photo.id, min_similarity)
            
            if group_members:
                # Mark all members as processed
                for member in group_members:
                    processed_ids.add(member['photo_id'])
                
                # Calculate average similarity and quality scores
                avg_similarity = sum(m['score'] for m in group_members) / len(group_members)
                avg_quality = sum(m.get('metadata', {}).get('file_size', 0) for m in group_members) / len(group_members) if group_members else 0
                
                # Apply min_similarity and min_quality filters
                if avg_similarity < min_similarity or avg_quality < min_quality:
                    continue
                
                group = {
                    'group_id': str(photo.id),
                    'member_count': len(group_members),
                    'similarity_score': avg_similarity,
                    'quality_score': avg_quality,
                    'avg_similarity': avg_similarity,
                    'members': group_members[:limit],  # Limit members per group
                    'metadata': {
                        'created_at': photo.uploaded_at.isoformat() if photo.uploaded_at else None,
                    }
                }
                groups.append(group)
        
        # Sort groups by specified criterion and direction
        reverse_sort = sort_order == 'desc'
        if sort_by == 'similarity':
            groups.sort(key=lambda g: g['similarity_score'], reverse=reverse_sort)
        elif sort_by == 'quality':
            # Sort by average quality (file_size proxy)
            groups.sort(key=lambda g: g['quality_score'], reverse=reverse_sort)
        
        # Apply pagination
        return groups[skip : skip + limit]
