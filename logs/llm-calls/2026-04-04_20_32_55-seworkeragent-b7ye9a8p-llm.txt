class SimilarityGroupService:
    """Service for querying similarity groups from Qdrant with pagination and filtering."""
    
    def __init__(self, db: Session):
        """Initialize service with database session and Qdrant client.

            db: SQLAlchemy database session
        """

    def list_groups(
        self,
        skip: int = 0,
        limit: int = 10,
        sort_by: str = "similarity",
        sort_order: str = "desc",
        min_similarity: float = 0.0,
        min_quality: float = 0.0
    ) -> List[Dict]:
        """List similarity groups with pagination, filtering, and sorting.
        
        Args:
            skip: Number of groups to skip (pagination offset)
            limit: Maximum number of groups to return
            sort_by: 'similarity' (by score) or 'quality' (by resolution/file_size)
            sort_order: 'asc' or 'desc' for sort direction
            min_similarity: Minimum similarity score threshold (0.0-1.0)
            min_quality: Minimum quality score threshold (0.0-1.0)
        
        Returns:
            List of similarity groups with metadata and thumbnails
        """
        # Fetch all photos with embeddings from database
        # Filter by min_similarity at query level for efficiency
        photos = self.db.query(Photo).all()
        if not photos:
            return []
        
        # Build groups by querying Qdrant for each photo's embedding
        groups = []
        processed_ids = set()
        
        for photo in photos:
            if photo.id in processed_ids:
                continue
            
            # Get embedding for this photo
            embedding = self.db.query(Embedding).filter(
                Embedding.photo_id == photo.id
            ).first()
            
            if not embedding or not embedding.qdrant_point_id:
                continue
            
            # Query Qdrant for similar photos (similarity group)
            # Note: In production, would fetch actual vector from Qdrant
            # For now, use a placeholder that would be replaced with real vector
            group_members = self._get_group_members(photo.id, min_similarity)
            
            if group_members:
                # Mark all members as processed
                for member in group_members:
                    processed_ids.add(member['photo_id'])
                
                # Calculate average similarity and quality scores
                avg_similarity = sum(m['score'] for m in group_members) / len(group_members)
                avg_quality = sum(m.get('metadata', {}).get('file_size', 0) for m in group_members) / len(group_members) if group_members else 0
                
                # Apply min_similarity and min_quality filters
                if avg_similarity < min_similarity or avg_quality < min_quality:
                    continue
                
                group = {
                    'group_id': str(photo.id),
                    'member_count': len(group_members),
                    'similarity_score': avg_similarity,
                    'quality_score': avg_quality,
                    'avg_similarity': avg_similarity,
                    'members': group_members[:limit],  # Limit members per group
                    'metadata': {
                        'created_at': photo.uploaded_at.isoformat() if photo.uploaded_at else None,
                    }
                }
                groups.append(group)
        
        # Sort groups by specified criterion and direction
        reverse_sort = sort_order == 'desc'
        if sort_by == 'similarity':
            groups.sort(key=lambda g: g['similarity_score'], reverse=reverse_sort)
        elif sort_by == 'quality':
            # Sort by average quality (file_size proxy)

    Scans folder for new/modified/deleted photos, queues changes for processing.
    Returns job_id for tracking progress via WebSocket.
    """
    if job_queue_manager is None:
        return JSONResponse(status_code=503, content={"error": "Job queue not initialized"})
    
    if not folder_path:
        folder_path = os.getenv("PHOTOS_FOLDER", "./photos")
    
    if not os.path.isdir(folder_path):
        return JSONResponse(status_code=400, content={"error": f"Folder not found: {folder_path}"})
    
    try:
        # Initialize scanner and database session
        scanner = FolderScanner()
        session = job_queue_manager.SessionLocal()
        
        # Scan folder for changes (new, modified, deleted photos)
        photo_ids, change_count = scanner.scan_folder(folder_path, session)
        session.close()
        
        if change_count == 0:
            return JSONResponse(status_code=200, content={
                "message": "No changes detected",
                "changes_found": 0
            })
        
        # Create job for processing changed photos
        job_id = str(uuid.uuid4())
        job_created = job_queue_manager.create_job(job_id, change_count)
        
        if not job_created:
            return JSONResponse(status_code=500, content={"error": "Failed to create processing job"})
        
        # Queue photos for processing
        for photo_id in photo_ids:
            asyncio.create_task(job_queue_manager.process_photo(job_id, photo_id))
        
        return JSONResponse(status_code=202, content={
            "job_id": job_id,
            "message": "Rescan initiated",
            "changes_found": change_count,
            "photos_queued": len(photo_ids)
        })
    except Exception as e:
        print(f"Error during rescan: {e}")
        return JSONResponse(status_code=500, content={"error": str(e)})
