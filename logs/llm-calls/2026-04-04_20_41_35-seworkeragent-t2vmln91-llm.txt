     1	/**
     2	 * Structured logging utility for React frontend.
     3	 * Provides Winston-compatible logging with performance metrics and error tracking.
     4	 */
     5	
     6	const LOG_LEVELS = {
     7	  DEBUG: 0,
     8	  INFO: 1,
     9	  WARNING: 2,
    10	  ERROR: 3,
    11	  CRITICAL: 4
    12	};
    13	
    14	class StructuredLogger {
    15	  constructor(name, logLevel = 'INFO') {
    16	    this.name = name;
    17	    this.logLevel = LOG_LEVELS[logLevel.toUpperCase()] || LOG_LEVELS.INFO;
    18	    this.logs = [];
    19	  }
    20	
    21	  /**
    22	   * Format log entry as structured JSON with timestamp and context.
    23	   */
    24	  formatLogEntry(level, message, extra = {}) {
    25	    return {
    26	      timestamp: new Date().toISOString(),
    27	      level,
    28	      logger: this.name,
    29	      message,
    30	      ...extra
    31	    };
    32	  }
    33	
    34	  /**
    35	   * Log a message at the specified level.
    36	   */
    37	  log(level, message, extra = {}) {
    38	    const levelValue = LOG_LEVELS[level.toUpperCase()] || LOG_LEVELS.INFO;
    39	    if (levelValue < this.logLevel) {
    40	      return;
    41	    }
    42	
    43	    const logEntry = this.formatLogEntry(level, message, extra);
    44	    this.logs.push(logEntry);
    45	
    46	    // Output to console with appropriate method
    47	    const consoleMethod = level.toLowerCase() === 'critical' ? 'error' : level.toLowerCase();
    48	    console[consoleMethod](JSON.stringify(logEntry));
    49	
    50	    // Send to backend for aggregation
    51	    this.sendToBackend(logEntry);
    52	  }
    53	
    54	  /**
    55	   * Send log entry to backend for aggregation and monitoring.
    56	   */
    57	  sendToBackend(logEntry) {
    58	    // Use beacon API for reliability (doesn't block page unload)
    59	    if (navigator.sendBeacon) {
    60	      navigator.sendBeacon('/api/logs', JSON.stringify(logEntry));
    61	    } else {
    62	      // Fallback to fetch with keepalive
    63	      fetch('/api/logs', {
    64	        method: 'POST',
    65	        body: JSON.stringify(logEntry),
    66	        keepalive: true,
    67	        headers: { 'Content-Type': 'application/json' }
    68	      }).catch(() => {
    69	        // Silently fail if backend is unavailable
    70	      });
    71	    }
    72	  }
    73	
    74	  debug(message, extra = {}) {
    75	    this.log('DEBUG', message, extra);
    76	  }
    77	
    78	  info(message, extra = {}) {
    79	    this.log('INFO', message, extra);
    80	  }
    81	
    82	  warning(message, extra = {}) {
    83	    this.log('WARNING', message, extra);
    84	  }
    85	
    86	  error(message, extra = {}) {

    88	  }
    89

    92	  }
    93	
    94	  /**

    97	  logPerformance(operation, durationMs, extra = {}) {

   103	  }
   104
