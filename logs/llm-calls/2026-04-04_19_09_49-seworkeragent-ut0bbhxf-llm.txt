     9	  const debounceTimerRef = useRef(null);

    11	
    12	  const handleInputChange = useCallback((e) => {
    13	    const newQuery = e.target.value;

    33	  }, [query, onSearch]);
    34	
    35	  useEffect(() => {
    36	    return () => {
    37	      if (debounceTimerRef.current) {
    38	        clearTimeout(debounceTimerRef.current);
    39	      }
    40	    };
    41	  }, []);
    42	
    43	  return (
    44	    <form className="SearchBox" onSubmit={handleSubmit}>
    45	      <input

    14	        self._cache_hits = 0
    15	        self._cache_misses = 0
    16	
    17	    def encode(self, texts, batch_size=32):
    18	        """Encode texts to embeddings with caching and batching.
    19	        
    20	        Args:
    21	            texts: List of strings or single string
    22	            batch_size: Batch size for encoding (M1 optimized default)
    23	        
    24	        Returns:
    25	            numpy array of embeddings
    26	        """
    27	        if isinstance(texts, str):
    28	            texts = [texts]
    29	        
    30	        # Check cache for exact matches
    31	        uncached = []
    32	        uncached_indices = []
    33	        embeddings = [None] * len(texts)
    34	        
    35	        for i, text in enumerate(texts):
    36	            if text in self._embedding_cache:
    37	                embeddings[i] = self._embedding_cache[text]
    38	                self._cache_hits += 1
    39	            else:
    40	                uncached.append(text)
    41	                uncached_indices.append(i)
    42	                self._cache_misses += 1
    43	        
    44	        # Encode uncached texts in batches
    45	        if uncached:
    46	            new_embeddings = self.model.encode(
    47	                uncached,
    48	                batch_size=batch_size,
    49	                convert_to_numpy=True,
    50	                show_progress_bar=False
    51	            )
    52	            
    53	            for idx, emb in zip(uncached_indices, new_embeddings):
    54	                embeddings[idx] = emb
    55	                self._embedding_cache[uncached[uncached_indices.index(idx)]] = emb
    56	        
    57	        return np.array(embeddings)

    64	            'hits': self._cache_hits,
    65	            'misses': self._cache_misses,
    66	            'hit_rate': hit_rate,
